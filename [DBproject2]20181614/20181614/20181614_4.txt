select T.id from (select shipment.package_id as id, shipment.date, shipment.month, shipment.year from package, shipment, receipt_package where package.package_id = receipt_package.package_id and receipt_package.package_id = shipment.package_id and shipment.place_type = "Truck" and shipment.place_number = "1721" and available = "Yes") as T order by T.year desc, T.month desc, T.date desc limit 1
